    let x: i32;  // 声明 x, 不需要使用 mut 修饰。

        x = 100;  // 初始化 x, 不需要 x 是 mut 的, 因为这是初始化，不是修改。

fn variable_bound1() {
    let mut v = Vec::new();  // 此处 v 必须用 mut 修饰，代表可写。
    v.push(1);
    v.push(2);
    v.push(3);

    let v = v;  // 只读。
    for i in &v {
        println!("{}", i)
    }
}

fn variable_bound2() {
    let v = Vec::new();
    let mut v = v;
    v.push(1);
    v.push(2);
    println!("{:?}", v);
}

fn type_td1() {
    let elem = 5u8;  // 未明确标出变量类型为，但通过字面量后缀，编译器能知道变量 elem 的类型为 u8。
    let mut vec = Vec::new();
    vec.push(elem);  // 此处编译器可以推导出vec的实际类型是是 Vec<u8>
    println!("{:?}", vec);
}
